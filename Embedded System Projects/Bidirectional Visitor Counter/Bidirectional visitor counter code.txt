#include <LiquidCrystal.h>

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

int sensor1 = 8;
int sensor2 = 9;
int relay = 10;

int count = 0;

void setup() {
  pinMode(sensor1, INPUT);
  pinMode(sensor2, INPUT);
  pinMode(relay, OUTPUT);

  digitalWrite(relay, LOW);

  lcd.begin(16, 2);
  lcd.print("Visitor Count");
  delay(2000);
  lcd.clear();
}

void loop() {

  // Person Enter
  if (digitalRead(sensor1) == LOW) {
    delay(300);
    if (digitalRead(sensor2) == LOW) {
      count++;
      while (digitalRead(sensor2) == LOW);
    }
  }

  // Person Exit
  if (digitalRead(sensor2) == LOW) {
    delay(300);
    if (digitalRead(sensor1) == LOW) {
      if (count > 0)
        count--;
      while (digitalRead(sensor1) == LOW);
    }
  }

  lcd.setCursor(0, 0);
  lcd.print("Visitors: ");
  lcd.print(count);
  lcd.print("   ");

  if (count > 0) {
    digitalWrite(relay, HIGH); // Light ON
    lcd.setCursor(0, 1);
    lcd.print("Light: ON ");
  } else {
    digitalWrite(relay, LOW);  // Light OFF
    lcd.setCursor(0, 1);
    lcd.print("Light: OFF");
  }

  delay(100);
}
